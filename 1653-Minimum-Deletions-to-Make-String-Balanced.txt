class Solution 
{
public:
    int minimumDeletions(string s) 
    {
        int n = s.size(), count = 0;
        stack <char> st;
        for(char ch : s)
        {
            if(!st.empty() && st.top() == 'b' && ch == 'a') 
            {
                st.pop();
                count++;
            }
            else st.push(ch);
        }
        return count;
    }
};
